Banker's Algorithm

Posted on 2016-11-23

实现思路

每个新进程在加入系统时，必须申明在运行中可能需要的每种资源类型的最大单元数目，其数目不应超过系统所拥有资源的总量。当进程请求一组资源时，系统必须首先确定是否有足够的资源分配给该进程。若可以分配，需进一步计算在将这些资源分配给进程后，是否会使系统处于不安全状态。如果不会才分配资源，否则让进程等待。

需要定义的数据结构

  1. 可利用资源向量Available，是个含有m个元素的数组，其中的每一个元素代表一类可利用的资源数目。如果Available[j]=K，则表示系统中现有Rj类资源K个。
  2. 最大需求矩阵Max，这是一个n×m的矩阵，它定义了系统中n个进程中的每一个进程对m类资源的最大需求。如果Max[i,j]=K，则表示进程i需要Rj类资源的最大数目为K。
  3. 分配矩阵Allocation，这也是一个n×m的矩阵，它定义了系统中每一类资源当前已分配给每一进程的资源数。如果Allocation[i,j]=K，则表示进程i当前已分得Rj类资源的数目为K。
  4. 需求矩阵Need，这也是一个n×m的矩阵，用以表示每一个进程尚需的各类资源数。如果Need[i,j]=K，则表示进程i还需要Rj类资源K个，方能完成其任务。
  5. 矩阵间的关系：Need[i,j]=Max[i,j]-Allocation[i,j]

银行家算法

设进程P提出请求Request[i]，则银行家算法按如下规则进行判断。
(1)如果Request[P][i] <= Need[P][i]，则转（2)；否则，出错。
(2)如果Request[P][i] <= Available[i]，则转（3)；否则，等待。
(3)系统试探分配资源，修改相关数据：
    Available[i] -= Request[P][i];
    Allocation[P][i] += Request[P][i];
    Need[P][i] -= Request[P][i];
(4)系统执行安全性检查，如安全，则分配成立；否则试探险性分配作废，系统恢复原状，进程等待。

安全性检查

(1)设置两个工作向量 Work = Available; Finish = {false...}
(2)从进程集合中找到一个满足下述条件的进程，
    Finish == false;
    Need <= Work;
    如找到，执行(3)；否则，执行(4)
(3)设进程获得资源，可顺利执行，直至完成，从而释放资源。
    Work = Work + Allocation;
    Finish = true;
    返回(2);
(4)如所有的进程Finish = true，则表示安全；否则系统不安全。

#include<iostream>
#include<stack>
#include<string.h>
#include<windows.h>
#define P 100  // 最大进程数
#define R 50   // 最大资源类数
using namespace std;

int Available[R], Max[P][R], Allocation[P][R], Need[P][R], Request[R];
int TempAvailable[R], TempAllocation[P][R], TempNeed[P][R];
int NumP = 0, NumR = 0;
bool safety; //安全状态
void check();

void init() {
    memset(Available, 0, sizeof(Available));
    memset(TempAvailable, 0, sizeof(TempAvailable));
    memset(Max, 0, sizeof(Max));
    memset(Allocation, 0, sizeof(Allocation));
    memset(TempAllocation, 0, sizeof(TempAllocation));
    memset(Need, 0, sizeof(Need));
    memset(TempNeed, 0, sizeof(TempNeed));
    memset(Request, 0, sizeof(Request));

    cout << "输入进程个数和资源个数：" << endl;
    cin >> NumP >> NumR;
    cout << "输入各进程最大资源需求(Max矩阵)：" << endl;
    for (int i = 0; i < NumP; i++) {
        for (int j = 0; j < NumR; j++) {
            cin >> Max[i][j];
        }
    }
    cout << "输入各进程已分配资源(Allocation矩阵)：" << endl;
    for (int i = 0; i < NumP; i++) {
        for (int j = 0; j < NumR; j++) {
            cin >> Allocation[i][j];
        }
    }
    cout << "输入可分配资源数(Available向量)：" << endl;
    for (int j = 0; j < NumR; j++) {
        cin >> Available[j];
    }
    for (int i = 0; i < NumP; i++) {
        for (int j = 0; j < NumR; j++) {
            Need[i][j] = Max[i][j] - Allocation[i][j];
        }
    }
}

void bank(int pid, int request[R]) {
    bool flag = true;
    for (int j = 0; j < NumR; j++) {
        if (request[j] > Need[pid][j] || request[j] > Available[j]) {
            cout << "请求资源数量超过进程需求量or系统可用资源数" << endl;
            flag = false;
        }
    }
    if (flag) {
        //保存当前各矩阵的值
        memcpy(TempAvailable, Available, sizeof(TempAvailable));
        memcpy(TempAllocation, Allocation, sizeof(TempAllocation));
        memcpy(TempNeed, Need, sizeof(TempNeed));

        for (int j = 0; j < NumR; j++) {
            Available[j] -= request[j];
            Allocation[pid][j] += request[j];
            Need[pid][j] -= request[j];
        }
        check();
    }
    Sleep(1000);
}

void check() {
    stack<int> stk;
    safety = true;
    int k = 0; //已成功处理的进程数
    bool have; //记录某次遍历是否有入栈操作
    bool found = false; //是否找到满足条件的进程
    int XL[P] = { 0 }; int v = 0; //记录序列
    int Work[P];
    bool Finish[P];
    // 初始化Work[]和Finish[]
    memcpy(Work, Available, sizeof(Work));
    for (int i = 0; i < P; i++) {
        Finish[i] = false;
    }

    do {
        have = false;
        for (int i = 0; i < NumP; i++) {
            if (!Finish[i]) {
                found = true;
                for (int j = 0; j < NumR; j++) {
                    if (Need[i][j] > Work[j]) {
                        found = false;
                    }
                }
                if (found) {
                    //入栈
                    stk.push(i);
                    have = true;
                    found = false;
                }
            }
        }
        if (!have) {
            if (k == 0) {
                //从没有加入过
                break;
            }
            //回退
            v--;
            int id = XL[v];
            XL[v] = 0;
            Finish[id] = false;
            for (int j = 0; j < NumR; j++) {
                Work[j] -= Allocation[id][j];
            }
            k--;
        }
        if (!stk.empty()) {
            int id = stk.top();
            stk.pop();
            Finish[id] = true;
            for (int j = 0; j < NumR; j++) {
                Work[j] += Allocation[id][j];
            }
            k++;
            XL[v++] = id;
        }
    } while (k < NumP);

    if (k == 0) {
        safety = false;
        cout << "系统处于不安全状态" << endl;
    } else {
        safety = true;
        cout << "安全序列为：";
        for (int i = 0; i < NumP; i++) {
            cout << "P" << XL[i] << "  ";
        }
        cout << endl;
    }
}

void status() {
    cout << endl << "------------------------------------------------" << endl;
    int i, j;
    cout << "当前状态：" << endl;
    cout << "PID    Max    Allocation    Need" << endl;
    for (i = 0; i < NumP; i++) {
        cout << i << "     ";
        for (j = 0; j < NumR; j++) {
            cout << Max[i][j] << "  ";
        }
        cout << "  |  ";
        for (j = 0; j < NumR; j++) {
            cout << Allocation[i][j] << "  ";
        }
        cout << "  |  ";
        for (j = 0; j < NumR; j++) {
            cout << Need[i][j] << "  ";
        }
        cout << endl;
    }
    cout << "系统可用资源：" << endl;
    for (j = 0; j < NumR; j++) {
        cout << Available[j] << "  ";
    }
    cout << endl;
    cout << endl << "------------------------------------------------" << endl;
}

int main()
{
    init();
    status();
    int PID = -1;
    while (true) {
        cout << "请输入进程ID:" << endl;
        cin >> PID;
        if (PID >= 0 && PID < NumP) {
            cout << "请输入请求资源数:" << endl;
            for (int j = 0; j < NumR; j++) {
                cin >> Request[j];
            }
            bank(PID, Request);
            if (!safety) {
                memcpy(Available, TempAvailable, sizeof(Available));
                memcpy(Allocation, TempAllocation, sizeof(Allocation));
                memcpy(Need, TempNeed, sizeof(Need));
            }
            status();
        } else {
            cout << "进程ID不合法" << endl;
        }
    }
    return 0;
}
